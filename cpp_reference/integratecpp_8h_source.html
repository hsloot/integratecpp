687 template <typename UnaryRealFunction_>
688 inline integrator::return_type integrator::operator()(UnaryRealFunction_ &&fn,
689  double lower,
690  double upper) const {
691  static_assert(
692  type_traits::is_invocable_r<
693  double, typename std::remove_reference<UnaryRealFunction_>::type,
694  const double>::value,
695  "`UnaryRealFunction_` is not invocable with `const double` and return "
696  "value `double`");
697 
698  // NOTE: check validity of configuration parameters
699  const auto throw_if_invalid_config = [](const config_type config) {
700  if (config.max_subdivisions <= 0) {
701  throw invalid_input_error("the input is invalid");
702  } else if (config.absolute_accuracy <= 0. &&
703  config.relative_accuracy <
704  std::max(50. * std::numeric_limits<double>::epsilon(),
705  0.5e-28)) {
706  throw invalid_input_error("the input is invalid");
707  } else if (config.work_size < 4 * config.max_subdivisions) {
708  throw invalid_input_error("the input is invalid");
709  } else {
710  return;
711  }
712  };
713  throw_if_invalid_config(config_);
714  // NOTE: check validity of bounds
715  const auto throw_if_invalid_bounds = [](const double lower,
716  const double upper) {
717  if (std::isnan(lower) || std::isnan(upper)) {
718  throw invalid_input_error("the input is invalid");
719  } else {
720  return;
721  }
722  };
723  throw_if_invalid_bounds(lower, upper);
724 
725  // NOTE: create local copies for input variables and references to an
726  // instance of output variables (as `Rdqag[si]` interface requires pointers to
727  // non-const variables). use names as described in the API of `Rdqag[is]`.
728  auto limit = config_.max_subdivisions;
729  auto epsrel = config_.relative_accuracy;
730  auto epsabs = config_.absolute_accuracy;
731  auto lenw = config_.work_size;
732 
733  auto out = return_type{}; // NOTE: construct returned object
734  auto &result = out.value;
735  auto &abserr = out.absolute_error;
736  auto &last = out.subdivisions;
737  auto &neval = out.neval;
738 
739  // NOTE: create variable for error code of `Rdqag[si]`
740  auto ier = 0;
741 
742  // NOTE: create working array and index array
743  auto iwork = std::vector<int>(limit);
744  auto work = std::vector<double>(lenw);
745 
746  // NOTE: create non-capturing callback Lambda (which can be implicitely
747  // converted to a fnct.-pointer of signature `integr_fn` aka
748  // `void(double *, int, void *)`).
749  // the actual integrand function is passed through the `void *` in the last
750  // argument alongside with a `std::unique_ptr` to capture exceptions during
751  // function evaluations.
752  const auto integrand_callback = [](double *x, int n, void *ex) {
753  using iterator = double *;
754  using const_iterator = const double *;
755  const auto cbegin = [](const double *x) {
756  return static_cast<const_iterator>(&x[0]);
757  };
758  const auto cend = [](const double *x, const int n) {
759  return static_cast<const_iterator>(&x[n]);
760  };
761  const auto begin = [](double *x) { return static_cast<iterator>(&x[0]); };
762  using ex_t =
763  std::pair<typename std::remove_reference<UnaryRealFunction_>::type,
764  std::exception_ptr>;
765 
766  auto &fn_integrand = (*static_cast<ex_t *>(ex)).first;
767  auto &e_ptr = (*static_cast<ex_t *>(ex)).second;
768 
769  // NOTE: `guarded_transform` is a wrapper arround `std::transform`,
770  // catching all exceptions appart `std::bad_alloc` and storing them in the
771  // provided `std::exception_ptr`.
772  // an additional check is performed whether all results are finite.
773  // in case of errors, all function values are set to zero.
774  const auto guarded_transform =
775  [](const_iterator first, const_iterator last, iterator d_first,
776  typename std::remove_reference<UnaryRealFunction_>::type &fn,
777  std::exception_ptr &e_ptr) {
778  const auto cleanup = [](iterator first, std::size_t size) {
779  try {
780  std::fill_n(first, size, 0.);
781  } catch (...) {
782  }
783  };
784  try {
785  std::transform(first, last, d_first, fn);
786  } catch (const std::bad_alloc &e) {
787  // NOTE: memory allocation issues inside std::transform must not be
788  // ignored
789  std::rethrow_exception(std::current_exception());
790  } catch (const std::exception &e) {
791  cleanup(d_first, std::distance(first, last));
792  e_ptr = std::current_exception();
793  } catch (...) {
794  cleanup(d_first, std::distance(first, last));
795  e_ptr = std::make_exception_ptr(
796  integration_runtime_error("Unknown error"));
797  }
798 
799  if (!static_cast<bool>(e_ptr) &&
800  !std::all_of(d_first, d_first + std::distance(first, last),
801  [](const double x) { return std::isfinite(x); })) {
802  cleanup(d_first, std::distance(first, last));
803  e_ptr = std::make_exception_ptr(
804  integration_runtime_error("non-finite function value"));
805  }
806  };
807  guarded_transform(cbegin(x), cend(x, n), begin(x), fn_integrand, e_ptr);
808  };
809  auto ex = std::make_pair(std::forward<UnaryRealFunction_>(fn),
810  std::exception_ptr());
811  auto &e_ptr = ex.second;
812 
813  if (std::isfinite(lower) && std::isfinite(upper)) {
814  Rdqags(integrand_callback, &ex, &lower, &upper, &epsabs, &epsrel, &result,
815  &abserr, &neval, &ier, &limit, &lenw, &last, iwork.data(),
816  work.data());
817  } else {
818  // NOTE: boundary information requires a transformation for `Rdqagi`.
819  const auto translate_bounds = [](const double lower, const double upper) {
820  int inf;
821  double bound;
822  if (std::isfinite(lower)) {
823  inf = 1;
824  bound = lower;
825  } else if (std::isfinite(upper)) {
826  inf = -1;
827  bound = upper;
828  } else {
829  inf = 2;
830  bound = 0.;
831  }
832  return std::make_pair(bound, inf);
833  };
834  auto bounds_info = translate_bounds(lower, upper);
835  auto bound = std::move(bounds_info.first);
836  auto inf = std::move(bounds_info.second);
837 
838  Rdqagi(integrand_callback, &ex, &bound, &inf, &epsabs, &epsrel, &result,
839  &abserr, &neval, &ier, &limit, &lenw, &last, iwork.data(),
840  work.data());
841  }
842 
843  // NOTE: translate error codes from `Rdqag[is]` and evaluation errors from
844  // `fn` to suitable exceptions
845  const auto throw_if_error = [](const int error_code, std::exception_ptr e_ptr,
846  const return_type result) {
847  if (e_ptr) {
848  std::rethrow_exception(e_ptr);
849  }
850  if (error_code > 0) {
851  // NOTE: invalid argument errors should be caught during
852  // initialization
853  assert(error_code < 6);
854  if (error_code == 1) {
855  throw max_subdivision_error("maximum number of subdivisions reached",
856  result);
857 
858  } else if (error_code == 2) {
859  throw roundoff_error("roundoff error was detected", result);
860  } else if (error_code == 3) {
861  throw bad_integrand_error("extremely bad integrand behaviour", result);
862  } else if (error_code == 4) {
863  throw extrapolation_roundoff_error(
864  "roundoff error is detected in the extrapolation table", result);
865  } else if (error_code == 5) {
866  throw divergence_error("the integral is probably divergent", result);
867  } else {
868  throw std::logic_error( // # nocov
869  "invalid argument errors should be caught during "
870  "initialization"); // # nocov
871  }
872  }
873  return;
874  };
875  throw_if_error(ier, std::move(e_ptr), out);
876 
877  return out;
878 };
